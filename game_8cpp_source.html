27void setup_world_coordinates(int windowWidth, int windowHeight,
28 ShaderProgram &shader) {
29 const int scaleX = BOARD_WIDTH_PIXELS / GRID_WIDTH;
30 const int scaleY = BOARD_HEIGHT_PIXELS / GRID_HEIGHT;
31 const int offsetX = (windowWidth - BOARD_WIDTH_PIXELS) / 2;
32 const int offsetY = (windowHeight - BOARD_HEIGHT_PIXELS) / 2;
33 glm::mat4 view =
34 glm::translate(glm::mat4(1.0F), glm::vec3(offsetX, offsetY, 0.0F));
35 view = glm::scale(view, glm::vec3(scaleX, scaleY, 1.0F));
36 const glm::mat4 projection =
37 glm::ortho(0.0F, (float)windowWidth, 0.0F, (float)windowHeight);
38 shader_program_set_matrix4(shader, "projection", projection,
39 true); // Uses shader
40 shader_program_set_matrix4(shader, "view", view);
41 unbind_shader_program();
42}

114void Game::update(double deltaTime) {
115 fallTimer += deltaTime;
116 if (fallTimer < fallInterval) {
117 return;
118 }
119
120 auto validator = [&](const std::vector<TilePosition> &positions) {
121 return !board.CheckCollision(positions);
122 };
123
124 currentShape.Update(validator);
125 fallTimer = 0.0F;
126
127 auto shapeTiles = currentShape.GetTilePositions();
128 if (board.CheckShapeMovement(shapeTiles)) {
129 LOG_DEBUG("Shape landed");
130 auto modifiedRows = board.PlaceShape(shapeTiles);
131 int filledRow = board.CheckRows(modifiedRows);
132 while (filledRow != BOARD_NO_ROW_CLEARED) {
133 modifiedRows = board.UpdateBoard(filledRow);
134 filledRow = board.CheckRows(modifiedRows);
135 }
136 generateNewShape();

167void Game::generateNewShape() {
168
169 const int shapeIndex = distr(gen);
170 auto type = static_cast<ShapeType>(shapeIndex);
171
172 const int initialX = GRID_WIDTH / 2;
173 const int initialY = GRID_HEIGHT - 2; // top row (y increasing upwards)
174
175 // Construct the new shape and set currentShape
176 Shape newShape = Shape(type, initialX, initialY);
177 LOG_DEBUG("Shape generated {}", newShape.GetType());
178
179 if (newShape.GetType() == ShapeType::I) {
180 // I shape spawns one row higher
181 newShape = Shape(type, initialX, initialY - 1);
182 }
183
184 auto validator = [&](const std::vector<TilePosition>& positions) {
185 return !board.CheckCollision(positions);
186 };
187
188 if (!validator(newShape.GetTilePositions())) {
189 LOG_INFO("Game Over detected");
190 this->gameOver = true;
191 }
192
193 this->currentShape = std::move(newShape);
194
195}
